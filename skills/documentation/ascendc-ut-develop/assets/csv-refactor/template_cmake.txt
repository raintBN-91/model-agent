# ========== 说明 ==========

# 本文件用于架构子目录（如 arch32）的测试编译配置
# 如果测试在架构子目录下，需要创建此 CMakeLists.txt

# ========== 使用方法 ==========

# 1. 将此文件复制到 {算子目录}/tests/ut/op_host/arch32/CMakeLists.txt
# 2. 如果有多个架构（arch32, arch33），每个架构目录都需要独立的 CMakeLists.txt
# 3. UT_NAME 使用 OP_TILING_MODULE_NAME（Tiling 测试）或相应的模块名
# 4. MODE 通常为 PRIVATE

# ========== 模板内容 ==========

if(UT_TEST_ALL OR OP_HOST_UT)
    add_modules_ut_sources(UT_NAME ${OP_TILING_MODULE_NAME} MODE PRIVATE DIR ${CMAKE_CURRENT_SOURCE_DIR})
endif()
